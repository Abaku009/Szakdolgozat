Table users {
  user_id integer [primary key]
  first_name varchar
  last_name varchar
  email varchar
  hashed_password varchar
  created_at timestamp
  updated_at timestamp
  is_admin bool
}

Table films {
  film_id integer [primary key]
  title varchar
  price varchar
  format varchar
  film_language_id integer [ref: > films_language.film_language_id]
  film_category_id integer [ref: > films_category.film_category_id]
  film_storage_id integer [ref: > films_storage.film_storage_id]
}

Table films_language {
  film_language_id integer [primary key]
  language varchar

}

Table films_category {
  film_category_id integer [primary key]
  genre varchar
}

Table films_storage {
  film_storage_id integer [primary key]
  quantity integer
}

Table series_category {

Table series_storage {

  message_id integer [primary key]
